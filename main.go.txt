	block := Block{ This line initializes a new Block struct instance called block

// Get latest block
func (bc *Blockchain) GetLatestBlock() Block {
	return bc.Chain[len(bc.Chain)-1]
}
This function is responsible for getting the latest block in the blockchain. 
Since the blockchain is stored as a slice, we simply take the last element using length minus one

// Add block
func (bc *Blockchain) AddBlock(newBlock Block) { //Defines a method of the Blockchain struct that takes a new block and adds it to the chain.
	newBlock.PreviousHash = bc.GetLatestBlock().Hash //Sets the PreviousHash of the new block to the hash of the latest block in the blockchain.This creates the link between blocks
	newBlock.Hash = newBlock.CalculateHash() //Calculates the hash of the new block using its updated data (including the previous hash).
	bc.Chain = append(bc.Chain, newBlock) //Adds the new block to the end of the blockchain slice.
}

// Validate chain
func (bc *Blockchain) IsChainValid() bool { //Defines a method of the Blockchain struct that returns a boolean (true or false) indicating whether the blockchain is valid
	for i := 1; i < len(bc.Chain); i++ { // Loops through the blockchain starting from index 1 (skipping the genesis block), because the first block has no previous block to compare with
		current := bc.Chain[i] // Stores the current block being checked
		previous := bc.Chain[i-1] // Stores the previous block in the chain for comparison.

		if current.Hash != current.CalculateHash() { // Checks if the stored hash of the current block matches a newly calculated hash based on its contents.
			return false //If the hashes do not match, it means the block data has been tampered with, so the blockchain is invalid
		}

		if current.PreviousHash != previous.Hash { // Checks if the current block correctly references the hash of the previous block.
			return false // If the previous hash does not match, the chain is broken and invalid.
		}
	}
	return true // If all blocks pass validation, the blockchain is valid
}

// Calculate hash
func (b *Block) CalculateHash() string {
	dataBytes, _ := json.Marshal(b.Data)
	record := fmt.Sprintf("%d%s%s%s", b.Index, b.Timestamp, b.PreviousHash, string(dataBytes))
	hash := sha256.Sum256([]byte(record))
	return hex.EncodeToString(hash[:])
}

// Blockchain struct
type Blockchain struct {
	Chain []Block
}

// Create genesis block
func (bc *Blockchain) CreateGenesisBlock() Block {
	block := Block{
		Index:        0,
		Timestamp:    "01/06/2021",
		Data:         "Genesis Block",
		PreviousHash: "0",
	}
	block.Hash = block.CalculateHash()
	return block
}

// Get latest block
func (bc *Blockchain) GetLatestBlock() Block {
	return bc.Chain[len(bc.Chain)-1]
}

// Add block
func (bc *Blockchain) AddBlock(newBlock Block) {
	newBlock.PreviousHash = bc.GetLatestBlock().Hash
	newBlock.Hash = newBlock.CalculateHash()
	bc.Chain = append(bc.Chain, newBlock)
}

// Validate chain
func (bc *Blockchain) IsChainValid() bool {
	for i := 1; i < len(bc.Chain); i++ {
		current := bc.Chain[i]
		previous := bc.Chain[i-1]

		if current.Hash != current.CalculateHash() {
			return false
		}

		if current.PreviousHash != previous.Hash {
			return false
		}
	}
	return true
}

func main() {
	// Initialize blockchain
	fredio := Blockchain{}
	fredio.Chain = append(fredio.Chain, fredio.CreateGenesisBlock())

	// Add blocks
	fredio.AddBlock(Block{Index: 1, Timestamp: "02/06/2021", Data: map[string]int{"amount": 1000}})
	fredio.AddBlock(Block{Index: 2, Timestamp: "03/06/2021", Data: map[string]int{"amount": 2000}})

	// Print blockchain
	jsonData, _ := json.MarshalIndent(fredio, "", "  ")
	fmt.Println(string(jsonData))

	// Validate
	fmt.Println("Status Block", fredio.IsChainValid())

	// Tamper data
	fredio.Chain[1].Data = map[string]int{"amount": 100}

	// Validate again
	fmt.Println("Status Block", fredio.IsChainValid())
}
